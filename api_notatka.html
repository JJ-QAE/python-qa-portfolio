import requests

response = requests.get("https://api.example.com/posts/1")
print(response.status_code)  # 200
print(response.json())       # dane jako słownik
status codes
Kody odpowiedzi
Zawsze sprawdzaj status code w testach API. POST zwraca 201 a nie 200!
200  # OK ✅ — GET, PUT, DELETE
201  # Created ✅ — POST (nowy zasób)
404  # Not Found ❌
400  # Bad Request ❌ — złe dane
500  # Server Error ❌
Metody HTTP
GET
Pobierz dane
Pobiera dane z serwera. Najczęściej używana metoda.
response = requests.get(
    "https://jsonplaceholder.typicode.com/posts/1"
)
assert response.status_code == 200
assert "title" in response.json()
POST
Stwórz nowe dane
Wysyła nowe dane do serwera. Wymaga json=słownik. Zwraca 201!
new_post = {
    "title": "Test",
    "body": "Content",
    "userId": 1
}
response = requests.post(
    "https://jsonplaceholder.typicode.com/posts",
    json=new_post    # ← json= wysyła słownik!
)
assert response.status_code == 201   # POST → 201!
assert response.json()["title"] == "Test"
PUT
Zaktualizuj dane
Aktualizuje istniejący zasób. URL musi zawierać ID posta.
response = requests.put(
    "https://jsonplaceholder.typicode.com/posts/1",
    json={"title": "Nowy tytuł"}
)
assert response.status_code == 200
DELETE
Usuń dane
Usuwa zasób. Nie wysyłasz żadnych danych — tylko URL.
response = requests.delete(
    "https://jsonplaceholder.typicode.com/posts/1"
)
assert response.status_code == 200

# GET    → 200  | POST  → 201
# PUT    → 200  | DELETE → 200
Testowanie błędów
404 i pusta odpowiedź
Status code to za mało — sprawdzaj też ciało
Sam status code to za mało — sprawdzaj też ciało odpowiedzi. 404 powinien zwracać pusty obiekt {}.
# nieistniejący post → 404
def test_not_found():
    response = requests.get(
        "https://jsonplaceholder.typicode.com/posts/99999"
    )
    assert response.status_code == 404

# nieistniejący endpoint → 404
def test_wrong_endpoint():
    response = requests.get(
        "https://jsonplaceholder.typicode.com/doesntexist"
    )
    assert response.status_code == 404

# 404 + sprawdzamy że ciało jest puste
def test_empty_response():
    response = requests.get(
        "https://jsonplaceholder.typicode.com/posts/99999"
    )
    assert response.status_code == 404
    assert response.json() == {}   # puste ciało odpowiedzi
Mini projekt — kompletny zestaw testów
test_posts_api.py
Wszystko w jednym pliku
Kompletny zestaw testów dla jednego API — GET, POST, PUT, DELETE, 404, parametrize. fixture z conftest dla GET — bez importu. POST bez fixture — requests.post() bezpośrednio.
import pytest
import requests

# GET — lista wszystkich postów
def test_posts_api():
    response = requests.get("https://jsonplaceholder.typicode.com/posts")
    assert response.status_code == 200
    assert len(response.json()) == 100

# POST — json= wysyła słownik, status 201!
def test_new_post():
    new_post = {"title": "Test", "body": "Content", "userId": 1}
    response = requests.post(
        "https://jsonplaceholder.typicode.com/posts",
        json=new_post
    )
    assert response.status_code == 201
    assert response.json()["title"] == "Test"

# PUT i DELETE — zawsze z ID posta w URL
def test_update():
    response = requests.put("https://jsonplaceholder.typicode.com/posts/1")
    assert response.status_code == 200

def test_delete():
    response = requests.delete("https://jsonplaceholder.typicode.com/posts/1")
    assert response.status_code == 200

# 404
def test_not_found():
    response = requests.get("https://jsonplaceholder.typicode.com/posts/99999")
    assert response.status_code == 404

# parametrize + f-string
@pytest.mark.parametrize("post_id", [1, 2, 3])
def test_posts_exist(post_id):
    response = requests.get(f"https://jsonplaceholder.typicode.com/posts/{post_id}")
    assert response.status_code == 200
Złota zasada API testów: zawsze response = requests.get() — nie return. GET używa fixture z conftest. POST nie używa fixture — requests.post() bezpośrednio z json=słownik. POST zwraca 201 a nie 200. f"...{zmienna}" wstawia zmienną do URL.
